for (i in 1:length(seu_list)) {

  )
}

Correct Indexing of ATAC Fragments

Sample 2 had a distinct issue from the other samples where there were discordant number of cells in the Seurat object and fragment files. The steps below filters for common cells between the two and subset the data for indexing.

i=2

# Get all unique cells present in the fragment file
# (Warning: this can be slow for very large files)
library(Signac)


message("Processing sample: ", names(seu_list)[i])
  
  # 1. Fast extract unique barcodes from the fragment file using terminal tools
  # This uses 'zcat', 'cut' to get the 4th column, and 'uniq' to get the list
  cmd <- paste0("zcat < ", atac_fragments_file[i], " | cut -f4 | sort | uniq")
  cells_in_fragment_file <- system(cmd, intern = TRUE)
  
  # 2. Identify the overlap
  # IMPORTANT: Check if you need to add/remove prefixes here!
  # If Seurat has "Sample1_", use: intersect(gsub(".*_", "", colnames(seu_list[[i]])), cells_in_fragment_file)
  current_seurat_cells <- colnames(seu_list[[i]])
  common_cells <- intersect(current_seurat_cells, cells_in_fragment_file)
  
  message("Found ", length(common_cells), " overlapping cells out of ", length(current_seurat_cells))
  
  # 3. Subset the Seurat object to only these common cells
  seu_list[[i]] <- subset(seu_list[[i]], cells = common_cells)
  
  # 4. Clear old fragment data and add the new link
  Fragments(seu_list[[i]]) <- NULL
  Fragments(seu_list[[i]]) <- CreateFragmentObject(
    path = atac_fragments_file[i],
    cells = colnames(seu_list[[i]]), # Now they match perfectly
    validate = FALSE                 # Fast, since we just verified them manually
  )

Standardize ATAC peaks

library(GenomicRanges)

# 1. Get peaks from all objects
peaks_list <- lapply(seu_list, function(x) granges(x[["ATAC_MACS2"]]))

# 2. Combine and "reduce" (merge overlapping peaks into one)
combined_peaks <- reduce(unlist(as(peaks_list, "GRangesList")))

# 3. Quantify this new peak set in each sample
for (i in 1:length(seu_list)) {
  # Count fragments in the new unified peaks
  counts <- FeatureMatrix(
    fragments = Fragments(seu_list[[i]]),
    features = combined_peaks,
    cells = colnames(seu_list[[i]])
  )
  
  # Create a new Assay with the unified peaks
  seu_list[[i]][["ATAC"]] <- CreateChromatinAssay(
    counts = counts,
    fragments = Fragments(seu_list[[i]])
  )
}

meta <- liver_filtered@meta.data

meta <- meta |> 
  mutate(
    sennet_id = case_when(
      grepl("SN049H1", orig.ident) ~ "SNT323.JXVC.442",
      grepl("SN078H1", orig.ident) ~ "SNT575.PMHQ.397",
      grepl("SN087H1", orig.ident) ~ "SNT626.DNSL.944",
      grepl("SN091H1", orig.ident) ~ "SNT875.QRMN.232",
      TRUE ~ NA
      ),
    age = case_when(
      grepl("SN049H1", orig.ident) ~ 73,
      grepl("SN078H1", orig.ident) ~ 27,
      grepl("SN087H1", orig.ident) ~ 27,
      grepl("SN091H1", orig.ident) ~ 73,
      TRUE ~ NA
    ),
    sex = case_when(
      grepl("SN049H1", orig.ident) ~ "M",
      grepl("SN078H1", orig.ident) ~ "M",
      grepl("SN087H1", orig.ident) ~ "F",
      grepl("SN091H1", orig.ident) ~ "F",
      TRUE ~ NA
    ),
    age_group = case_when(
      age < 50 ~ "young",
      age > 50 ~ "old",
      TRUE ~ NA
    ),
    age_sex = glue::glue("{age}_{sex}")
  )
